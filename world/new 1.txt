# input loop
height = 0
while not (1 <= height <= 8):
   height = input("height: ")
# drawing loop
for h in range(height):
   line = str()
   ofst = height - h
   # render offset
   for s in range(ofst):
      line += ' '
   # render 1st half
   for b in range(h):
      line += '#'
   # render middle spacing
   for s in range(2):
      line += ' '
   # render 2nd half
   for b in range(h):
      line += '#'
   # console draw
   print(line)
